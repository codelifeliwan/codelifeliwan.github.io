void Calculator::button_dengyu_clicked(){

 
          
 
          
 S+="=";

 
          
 
          
// lineEditor->setText(S);

 
          
 
          
 switch(fuhao){

 
          
 
          
 case '+':result=num1+num2;break;

 
          
 
          
 case '-':result=num1-num2;break;

 
          
 
          
 case '*':result=num1*num2;break;

 
          
 
          
 case '/':if(num2)result=num1/num2;break;

 
          
 
          
 }

 
          
 
          
 int i=0;

 
          
 
          
 int result1=result;

 
          
 
          
 while(result1){

 
          
 
          
 i++;

 
          
 
          
 result1/=10;

 
          
 
          
 }

 
          
 
          
 for(;i;i--){

 
          
 
          
 S+=(result/(int)pow(10,i-1)+'0');

 
          
 
          
 result%=(int)pow(10,i-1);

 
          
 
          
 }

 
          
 
          
 lineEditor->setText(S);

 
          
 
          
 zhenghao=true;

 
          
 
          
 num1=num2=0;

 
          
 
          
 mark=1;

 
          
 
          
 S="";

 
          
 
          
}
