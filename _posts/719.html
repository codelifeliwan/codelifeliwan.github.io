#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#define maxrow 21
#define maxn 90000
#define maxhashsize 1000003
 
int begin[maxrow];
int end[maxrow]={0,3,4,3,0,5,6,5,0,1,2,1,0,7,8,7,0,9,10,9,0};
int a1[maxn][maxrow];
int a2[maxn][maxrow];
int fa1[maxn];
int fa2[maxn];
int steps1[maxn],steps2[maxn];
int path1[maxn],path2[maxn];
int head1[maxhashsize],next1[maxhashsize];
int head2[maxhashsize],next2[maxhashsize];
long long solution;
 
int hash(int t[]){
	int i,ret=0;
	for(i=0;i<21;i++){
		ret=ret*10+t[i];
		ret%=maxhashsize;
	}
	return ret;
}
 
int try_to_insert(int s,int t[][21],int mark){
	int h=hash(t[s]);
	int u;
	if(mark==1){
		u=head1[h];
	}else{
		u=head2[h];
	}
	while(u){
		if(memcmp(t[u],t[s],sizeof(t[0]))==0)return 0;
		if(mark==1)u=next1[u];
		else u=next2[u];
	}
	if(mark==1){
		next1[s]=head1[h];
		head1[h]=s;
	}else{
		next2[s]=head2[h];
		head2[h]=s;
	}
	return 1;
}
 
int check_isok(int s,int mark){
	int h,u;
	if(mark==1){
		h=hash(a1[s]);
		u=head2[h];
	}else{
		h=hash(a2[s]);
		u=head1[h];
	}
	while(u){
		if(mark==1){
			if(memcmp(a1[s],a2[u],sizeof(a1[s]))==0)
				return u;
			u=next2[u];
		}else{
			if(memcmp(a2[s],a1[u],sizeof(a2[s]))==0)
				return u;
			u=next1[u];
		}
	}
	return -1;
}
 
long long print1(int x1){
	if((fa1[x1])!=0)return print1(fa1[x1])*10+path1[x1];
	return 0;
}
 
long long print2(int x2,long long tmpsls){
	int tmp;
	while(x2!=1){
		if(path2[x2]==1)tmp=3;
		else if(path2[x2]==2)tmp=4;
		else if(path2[x2]==3)tmp=1;
		else if(path2[x2]==4)tmp=2;
		tmpsls=tmpsls*10+tmp;
		x2=fa2[x2];
	}
	return tmpsls;
}
 
void print_path(int x1,int x2){
	long long tmpsls=0;
	tmpsls=print2(x2,print1(x1));
	if(tmpsls<solution)
		solution=tmpsls;
}
 
int dbfs(){
	int front1=1,rear1=1,front2=1,rear2=1,mark,i,j;
	int tmp[maxrow],tmp1[2];
	int r=0;
	solution=9999999999999999;
	memset(a1,0,sizeof(a1));
	memset(a2,0,sizeof(a2));
	memcpy(a1[front1],begin,sizeof(begin));
	memcpy(a2[front2],end,sizeof(end));
	memset(head1,0,sizeof(head1));
	memset(head2,0,sizeof(head2));
	memset(next1,0,sizeof(next1));
	memset(next2,0,sizeof(next2));
	memset(fa1,0,sizeof(fa1));
	memset(fa2,0,sizeof(fa2));
	try_to_insert(1,a1,1);
	try_to_insert(1,a2,2);
	steps1[front1++]=0;
	steps2[front2++]=0;
	while(front1>rear1||front2>rear2){
		if(front1<=rear1)mark=2;
		else if(front2<=rear2)mark=1;
		else if(front1>front2)mark=2;
		else if(front1<=front2)mark=1;
		if(mark==1){
			if(steps1[rear1]>=8){
				if(rear1<front1)rear1++;
				continue;
			}
		}else{
			if(steps2[rear2]>=8){
				if(rear2<front2)rear2++;
				continue;
			}
		}
		for(i=1;i<=4;i++){
			if(mark==1){
				memcpy(tmp,a1[rear1],sizeof(tmp));
			}else{
				memcpy(tmp,a2[rear2],sizeof(tmp));
			}
			if(i==1){
				tmp1[0]=tmp[10];tmp1[1]=tmp[11];
				for(j=9;j>=0;j--)tmp[j+2]=tmp[j];
				tmp[0]=tmp1[0];tmp[1]=tmp1[1];
			}else if(i==2){
				tmp1[0]=tmp[9];tmp1[1]=tmp[10];
				for(j=11;j<21;j++)tmp[j-2]=tmp[j];
				tmp[19]=tmp1[0];tmp[20]=tmp1[1];
			}else if(i==3){
				tmp1[0]=tmp[0];tmp1[1]=tmp[1];
				for(j=2;j<12;j++)tmp[j-2]=tmp[j];
				tmp[10]=tmp1[0];tmp[11]=tmp1[1];
			}else if(i==4){
				tmp1[0]=tmp[19];tmp1[1]=tmp[20];
				for(j=18;j>8;j--)tmp[j+2]=tmp[j];
				tmp[9]=tmp1[0];tmp[10]=tmp1[1];
			}
			if(mark==1){
				memcpy(a1[front1],tmp,sizeof(tmp));
				steps1[front1]=steps1[rear1]+1;
				path1[front1]=i;
				if(try_to_insert(front1,a1,1)){
					fa1[front1]=rear1;
					front1++;
					if((j=check_isok(front1-1,1))!=-1){
						print_path(front1-1,j);
						r=1;
					}
				}
			}else{
				memcpy(a2[front2],tmp,sizeof(tmp));
				steps2[front2]=steps2[rear2]+1;
				path2[front2]=i;
				if(try_to_insert(front2,a2,2)){
					fa2[front2]=rear2;
					front2++;
					if((j=check_isok(front2-1,2))!=-1){
						print_path(j,front2-1);
						r=1;
					}
				}
			}
		}
		if(mark==1&&rear1<front1)rear1++;
		else if(mark==2&&rear2<front2)rear2++;
	}
	return r;
}
 
int main()
{
	freopen("test.txt","r",stdin);
	int NCASE,i,tmp[24];
	scanf("%d",&NCASE);
	while(NCASE--){
		for(i=0;i<24;i++)scanf("%d",&tmp[i]);
		for(i=0;i<21;i++)begin[i]=tmp[i];
		if(memcmp(begin,end,sizeof(end))==0){
			printf("PUZZLE ALREADY SOLVED\n");
		}else{
			if(!dbfs())printf("NO SOLUTION WAS FOUND IN 16 STEPS\n");
			else printf("%lld\n",solution);
		}
	}
	return 0;
}
