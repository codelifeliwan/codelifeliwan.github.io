Because the runner teaches algorithms at University, and he
has some hours to think while he is running, he has thought up a
programming problem which generalizes the problem of his sweat
pedometer. The problem consists of obtaining the minimum number of
active LEDs necessary to identify each one of the symbols, given a
number P of LEDs, and N symbols to be represented
with these LEDs (along with the codification of each symbol).

For example, in the previous sample P = 7 and
N = 10. Supposing the LEDs are numbered as:

“8” = 1 1 1 1 1 1 1; “9” = 1 1 1 1 0 1 1. In this case, LEDs 5 and

The

0s and 1s, with a space between them. A 1 means the corresponding
LED is part of the codification of the symbol. The maximum value of
P is 15 and the maximum value of N is 100. All
the symbols have different codifications.

The

number of active LEDs necessary to identify all the given
symbols.

代码：




#include<iostream>

#include<cstring>

#include<fstream>

using namespace std;

int map[105][20];

int vis[20];

int hang,lie;

int result;




void chuli(){

 for(int i=0;i<hang-1;i++){

 for(int j=i+1;j<hang;j++){

 bool found=false;

 for(int k=0;k<lie;k++){

 if(vis[k]&&map[i][k]!=map[j][k]){

 found=true;

 }

 }

 if(!found)return;

 }

 }

 int num=0;

 for(int i=0;i<lie;i++){

 if(vis[i])num++;

 }

 if(num<result)result=num;

}




void solve(int n){

 if(n==lie){

 chuli();

 }else{

 vis[n]=1;

 solve(n+1);

 vis[n]=0;

 solve(n+1);

 }

}




int main()

{

// freopen("test.txt","r",stdin);

 int ncase;

 cin>>ncase;

 while(ncase--){

 cin>>lie>>hang;

 for(int i=0;i<hang;i++){

 for(int j=0;j<lie;j++){

 cin>>map[i][j];

 }

 }

 memset(vis,0,sizeof(vis));

 result=lie;

 solve(0);

 cout<<result<<endl;

 }

 return 0;

}
