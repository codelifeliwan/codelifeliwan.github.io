#include<stdio.h>
struct point
{
double x;
double y;
};
struct sanjiaoxing
{
struct point p1;
struct point p2;
struct point p3;
};
struct point chuli(struct sanjiaoxing *san,long num)
{
struct point p;
double m1=0,m2=0,n1=0;
long i;
for(i=0;i<num;i++)
{
m1=m1+(((san[i].p1.x+san[i].p2.x+san[i].p3.x)/3)*((san[i].p1.x*san[i].p2.y+san[i].p1.y*san[i].p3.x+san[i].p2.x*san[i].p3.y-san[i].p2.y*san[i].p3.x-san[i].p1.y*san[i].p2.x-san[i].p1.x*san[i].p3.y)/2));
m2=m2+(san[i].p1.x*san[i].p2.y+san[i].p1.y*san[i].p3.x+san[i].p2.x*san[i].p3.y-san[i].p2.y*san[i].p3.x-san[i].p1.y*san[i].p2.x-san[i].p1.x*san[i].p3.y)/2;
n1=n1+(((san[i].p1.y+san[i].p2.y+san[i].p3.y)/3)*((san[i].p1.x*san[i].p2.y+san[i].p1.y*san[i].p3.x+san[i].p2.x*san[i].p3.y-san[i].p2.y*san[i].p3.x-san[i].p1.y*san[i].p2.x-san[i].p1.x*san[i].p3.y)/2));
}
p.x=m1/m2;
p.y=n1/m2;
return p;
}
int main()
{
struct sanjiaoxing san[1000];
struct point p[3];
long i,j,num;
freopen(“test.txt”,”r”,stdin);
while(scanf(“%ld”,&num)!=EOF&&0!=num)
{
if(1==num)
{
scanf(“%lf%lf”,&p[0].x,&p[0].y);
printf(“%.2lf %.2lfn”,p[0].x,p[0].y);
continue;
}
else if(2==num)
{
scanf(“%lf%lf”,&p[0].x,&p[0].y);
scanf(“%lf%lf”,&p[1].x,&p[1].y);
p[2].x=p[0].x+p[1].x;
p[2].y=p[0].y+p[1].y;
p[2].x=p[2].x/2;
p[2].y=p[2].y/2;
printf(“%.2lf %.2lfn”,p[2].x,p[2].y);
continue;
}
else
{
scanf(“%lf%lf”,&p[0].x,&p[0].y);
scanf(“%lf%lf”,&p[1].x,&p[1].y);
for(i=0;i<num-2;i++)
{
scanf(“%lf%lf”,&p[2].x,&p[2].y);
san[i].p1=p[0];
san[i].p2=p[1];
san[i].p3=p[2];
p[1]=p[2];
}
p[0]=chuli(san,num-2);
printf(“%.2lf %.2lfn”,p[0].x,p[0].y);
}
}
return 0;
}
