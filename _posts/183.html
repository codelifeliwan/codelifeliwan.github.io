常用方法看下面实例极其注释：


package RuntimeDemo001;

 


import java.io.IOException;

 


public class RuntimeDemo001 {


public static void main(String args[]){


Runtime run=Runtime.getRuntime();

//取得Runtime实例


long max=run.maxMemory();

//取得JVM最大内存


System.out.println(“JVM试图使用的最大内存量为：”+max);


long free=run.freeMemory();

//返回JVM剩余空间（字节）


System.out.println(“JVM空闲内存为：”+free);


long free1=run.freeMemory();

//返回JVM剩余空间（字节）


System.out.println(“JVM空闲内存为：”+free1);


long free2=run.freeMemory();

//返回JVM剩余空间（字节）


System.out.println(“JVM空闲内存为：”+free2);


String s=“jgscuysgdcdsytfdtyfs”;


for(int i=0;i<1000;i++){

//故意产生内存垃圾


s+=“sjbdcjshdbhsgvfdfy”;


}


long free3=run.freeMemory();

//返回JVM剩余空间（字节）


System.out.println(“JVM空闲内存为：”+free3);


long free4=run.freeMemory();

//返回JVM剩余空间（字节）


System.out.println(“JVM空闲内存为：”+free4);


run.gc();
//手动回收垃圾，释放空间


long free5=run.freeMemory();

//返回JVM剩余空间（字节）


System.out.println(“JVM空闲内存为：”+free5);


try {


Process pre=run.exec(“notepad”);

//执行本机可执行程序，赋值给进程对象


Thread.sleep(5000);

//进程暂停5s


pre.destroy();
//杀死进程


} catch (Exception e) {


e.printStackTrace();


}


}
