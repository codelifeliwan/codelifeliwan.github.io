#include<iostream>
#include<fstream>
#include<cmath>
#include<cstring>
using namespace std;
char Snum[50];
long solve()
{
long re=0;
int len=strlen(Snum);
int i=0;
while(len>=1)
{
re+=((Snum[i]-‘0′)*(pow(2,len)-1));
i++;
len–;
}
return re;
}
int main()
{
// freopen(“test.txt”,”r”,stdin);
while(cin>>Snum)
{
if(strcmp(“0″,Snum)==0)break;
long result=solve();
cout<<result<<endl;
}
return 0;
}
