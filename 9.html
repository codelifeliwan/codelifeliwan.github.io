判断三角形的内点个数

Posted on 19 一月, 2011 by wanli — No Comments ↓


记得有一次比赛就是考的这个，当时还没做出来，今天看书看到这个觉得比较有代表性，遂写此文。


题目就是给出很多点，坐标分别从1到99的二维矩阵，现在输入三角形的三个顶点的x，y坐标，要求输出在三角形内的点的个数，包括在三角形的边上的点。


此题就是使用三角形的面积，如果点在三角形内部，
则连接点到三角形的三个顶点，将大三角形分成三个小三角形，三个小三角形的面积之和等于大三角形的面积。而在三角形外部的点则不等于。


在此处三角形的面积求法有多种方法，有什么正弦公式，海伦公式等，但是那些误差太大，因为三角形的顶点坐标不一定是整数，所以不得不考虑误差问题。所以在这书上给了一个公式：


设三角形面积为A，三个顶点分别为(x0,y0),(x1,y1),(x2,y2)，则：

2A= =x0*y1+x2*y0+x1*y2-x2*y1-x0*y2-x1*y0


这个公式也有误差，但是比其他的乘方开方的误差稍微小了点。在使用公式的时候要考虑误差，就是在一定误差范围内认为它等于一个数。


下面看代码：

#include<stdio.h>

#include<math.h>

struct Point

{

        
double x;

        
double y;

};

 

double area2(struct Point
p0,struct Point p1,struct Point p2)

{

        
return
fabs(p0.x*p1.y+p2.x*p0.y+p1.x*p2.y-p2.x*p1.y-p0.x*p2.y-p1.x*p0.y);//求三角形面积函数

}

 

int main()

{

        
struct Point p[10000],sanjiao[3];

        
double area,area1;

        
int k,i;

        
freopen(“test.txt”,”r”,stdin);

        
for(i=0;i<=9800;i++)

        
{

                  
p[i].x=i�+1;

                  
p[i].y=i/99+1;

        
}

        
while(scanf(“%lf%lf%lf%lf%lf%lf”,&sanjiao[0].x,&sanjiao[0].y,&sanjiao[1].x,&sanjiao[1].y,&sanjiao[2].x,&sanjiao[2].y)!=EOF)

        
{

                  
k=0;

                  
area=area2(sanjiao[0],sanjiao[1],sanjiao[2]);

                  
for(i=0;i<=9800;i++)

                  
{

                           
area1=area2(sanjiao[1],sanjiao[2],p[i])+area2(sanjiao[0],sanjiao[2],p[i])+area2(sanjiao[0],sanjiao[1],p[i]);

                           
if(area1-area<=1e-9) //误差分析，在一定误差范围内认为相等,误差范围为10^-9

                                    
k++;

                  
}

                  
printf(“%dn”,k);

        
}

        
return 0;

}


这个是从第一个到最后一个挨个判断的，比较耗费时间，我们在中间加个判断，进行第一步过滤代码如下：

#include<math.h>

struct Point

{

        
double x;

        
double y;

};

 

double area2(struct Point
p0,struct Point p1,struct Point p2)

return

}

{

        
struct Point p[10000],sanjiao[3];

        
double area,area1;

        
int k,i;

        
freopen(“test.txt”,”r”,stdin);

        
for(i=0;i<=9800;i++)

{

                  
k=0;

                  
area=area2(sanjiao[0],sanjiao[1],sanjiao[2]);

                  
for(i=0;i<=9800;i++)

                  
{

                  
   
if((p[i].x>sanjiao[0].x&&p[i].x>sanjiao[1].x&&p[i].x>sanjiao[2].x)||(p[i].x<sanjiao[0].x&&p[i].x<sanjiao[1].x&&p[i].x<sanjiao[2].x))

}

                  
   
if((p[i].y>sanjiao[0].y&&p[i].y>sanjiao[1].y&&p[i].y>sanjiao[2].y)||(p[i].y<sanjiao[0].y&&p[i].y<sanjiao[1].y&&p[i].y<sanjiao[2].y))

continue;//第一步过滤

                  
   
}

误差范围为10^-9
