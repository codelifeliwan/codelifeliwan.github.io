//list.h
#ifndef LIST_H_
#define LIST_H_
#include<iostream>
#define OK 1
#define ERROR 0
typedef int Status;
using namespace std;
typedef struct node{
int number;
int bianhao;
struct node *next;
}ElemType;
typedef ElemType* nodePointer;
Status initList(nodePointer &head,nodePointer
&tail,int &num)
{
if(!num)return ERROR;
if(num)
{
int bianhao=1;
int number=num-1;
cout<<“Please enter the given
number of each people:”<<endl;
head=new ElemType;
nodePointer p=head;
cin>>p->number;
p->bianhao=bianhao++;
nodePointer p1=p;
p=p->next;
while(number)
{
int tempnumber;
cin>>tempnumber;
if(0==tempnumber)continue;
p=new ElemType;
p1->next=p;
p1=p;
p->number=tempnumber;
p->bianhao=bianhao++;
p=p->next;
number–;
}
tail=p1;
p1->next=head;
}
return OK;
}
Status list_Pop(nodePointer &head,nodePointer
&tail,int &num,int step)
{
if(!num)return ERROR;
if(NULL==head)return ERROR;
if(num==1){
nodePointer p=head;
cout<<p->bianhao<<endl;
delete head;
head=NULL;
return p->bianhao;
}
for(int i=1;i<step;i++)
tail=tail->next;
nodePointer p=tail->next;
head=p->next;
tail->next=head;
num–;
int re=p->number;
cout<<p->bianhao<<endl;
delete p;
return re;
}
#endif

#include<iostream>

#include”list.h”
using namespace std;

{
while(1)
{

}
}
}
