public String[] list()
//列出指定目录全部内容，只列出名字


public File[] listFiles()
//列出指定目录全部内容，包括路径


范例：


import java.io.*;

 


public class FileTest001


{


public static void main(String args[]) throws Exception{


File f=new File(“F:”+File.separator);


String S[]=f.list();


for(String X:S){


System.out.println(X);


}


File F[]=f.listFiles();


for(File X:F){


System.out.println(X);


}


}


}
