应用：

class A{

   
public void fun(){

      
System.out.println(“A方法调用“);

   
}

}

 

class B extends A{

   
public void fun(){

      
System.out.println(“B方法调用“);

   
}

}

 

class C extends A{

   
public void fun(){

      
System.out.println(“C方法调用“);

   
}

}

 

public class Main {

   
public static void main(String args[]){

      
fun(new A());

      
fun(new B());

      
fun(new C());

   
}

   

   
static void fun(A a){      
//Only once!

      
a.fun();

   
}

}
