第一题：HDOJ 1028
此题为一个DP，是背包问题方案数的问题，理论上有两种方法，一种可以使用记忆话递归搜索，另一种是直接递推，感觉两种方法都差不多，但是记忆话递归的超时，郁闷。。。。。
下面给出两种方法代码：
【一】：AC代码：
状态转移方程：V[j]=V[j-i],其中j表示“当前背包容量”，i表示现在计算的数字是i
#include<iostream>
#include<cstring>
#include<algorithm>
#include<fstream>
#define maxn 125
using namespace std;
int V[maxn];
int num;
int main()
{
freopen(“test.txt”,”r”,stdin);
while(cin>>num){
memset(V,0,sizeof(V));
V[0]=1;
for(int i=1;i<=num;i++){
for(int j=i;j<=num;j++){
V[j]+=V[j-i];
}
}
cout<<V[num]<<endl;
}
return 0;
}
【二】：TLE代码：
状态转移方程：V[i][j]=sum(V[n-x][x]);
#include<iostream>
#include<cstring>
#include<algorithm>
#include<fstream>
#define maxn 125
using namespace std;
int V[maxn*maxn][maxn];
int num;
int dp(int n,int m){
if(n==0)return 1;
int ans=V[n][m];
if(ans)return ans;
for(int i=m;i<=n;i++){
ans+=dp(n-i,i);
}
return ans;
}
int main()
{
// freopen(“test.txt”,”r”,stdin);
memset(V,0,sizeof(V));
while(cin>>num){
cout<<dp(num,1)<<endl;
}
return 0;
}

#include<iostream>
#include<cmath>
using namespace std;
struct point   //定义点坐标结构体
{
double x;
double y;
};
struct sideps //定义点到直线最短距离和最长距离结构体
{
double smax;
double smin;
};
double tp_d(point p1,point p2)//传入两点坐标返回两点间距离
{
return
sqrt((p1.x-p2.x)*(p1.x-p2.x)+(p1.y-p2.y)*(p1.y-p2.y));
}
double t_a(point p1,point p2,point
p3)//输入三点坐标，海伦公式求三角形面积
{
double side1=tp_d(p1,p2);
double side2=tp_d(p1,p3);
double side3=tp_d(p2,p3);
double ps=(side1+side2+side3)*0.5;
return
sqrt(ps*(ps-side1)*(ps-side2)*(ps-side3)); 
}
double in_pt(point p1,point p2,point p3)//算两向量内积
{
return (p2.x-p1.x)*(p3.x-p1.x)+(p2.y-p1.y)*(p3.y-p1.y);
}
sideps p_s_sxn(point p1,point p2,point
p3)//计算点到直线最短距离和最长距离
{
sideps smn;
smn.smax=tp_d(p1,p3)>tp_d(p1,p2)?tp_d(p1,p3):tp_d(p1,p2);
if(in_pt(p3,p1,p2)*in_pt(p2,p3,p1)>=0)
{
 
  smn.smin=t_a(p1,p2,p3)*2.0/tp_d(p2,p3);
}
else
{
 
  smn.smin=tp_d(p1,p3)<tp_d(p1,p2)?tp_d(p1,p3):tp_d(p1,p2);
}
return smn;
}
int main()
{
sideps s1,s2,s3,s4;
point p1,p2,p3;
double r,x1,x2,y1,y2;
int t;
cin>>t;
while(t–)
{
 
  cin>>p1.x>>p1.y>>r>>x1>>y1>>x2>>y2;
    p2.x=x1;p2.y=y1;
    p3.x=x1;p3.y=y2;
 
  s1=p_s_sxn(p1,p2,p3);
    p2.x=x1;p2.y=y2;
    p3.x=x2;p3.y=y2;
 
  s2=p_s_sxn(p1,p2,p3);
    p2.x=x2;p2.y=y2;

#include<iostream>
#include<cstring>
#include<algorithm>
#include<fstream>

using namespace std;
int M[maxn];
int a[maxn];
int num;
int b_search(int begin,int end,int x){
if(x<=a[begin])return 0;
int m;
while(begin<end){
m=(begin+end)/2;
if(a[m-1]<x&&a[m]>=x)return
m;
else if(a[m]<x)begin=m;
else end=m;
}
return end;
}
int main()
{
// freopen(“test.txt”,”r”,stdin);

for(int i=0;i<num;i++){
cin>>M[i];
}
a[0]=M[0];
int ans=1;
for(int i=1;i<num;i++){
if(M[i]>a[ans-1]){
a[ans]=M[i];
ans++;
}else{
int r=b_search(0,ans,M[i]);
a[r]=M[i];
}
}
cout<<ans<<endl;
}
return 0;
}

void dijkstra(int node)
{
   
memset(vist,0,sizeof(vist));
    for(int
i=1;i<=N;i++)
     
  dist[i]=inf;
    dist[node]=0;
    for(int
i=1;i<=N;i++)
    {
     
  int x=0,m=inf;
     
  for(int y=1;y<=N;y++)
     
     
if(!vist[y]&&dist[y]<m)
     
      {
     
     
    x=y;
     
     
    m=dist[y];
     
      }

break;

  for(int y=1;y<=N;y++)
     
     
dist[y]=min(dist[y],dist[x]+map[x][y]);
    }
}

{
    if(dp[node]!=-1)
     
  return dp[node];
    if(node==2)
